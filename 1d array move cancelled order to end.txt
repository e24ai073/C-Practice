#include <stdio.h>
int main() {
    int N;
    printf("Enter number of orders: ");
    scanf("%d", &N);
    int orders[N];
    printf("Enter order IDs:\n");
    for (int i = 0; i < N; i++) {
        scanf("%d", &orders[i]);
    }
    int index = 0;  
    for (int i = 0; i < N; i++) {
        if (orders[i] != 0) {
            orders[index++] = orders[i];
        }
}
    while (index < N) {
        orders[index++] = 0;
    }
   printf("Updated order list:\n");
    for (int i = 0; i < N; i++) {
        printf("%d ", orders[i]);
    }

    return 0;
}
